	<script type="text/javascript">
		var currIndex = 0,
			tablist = $('.list_subnav ul li');
			panellist = $('.box5 .list_con');
		tablist.each(function(index,elem){
			$(elem).attr('index',index);
		});
		tablist.mouseover(function(event){
			var index = $(event.currentTarget).attr('index');
			$(tablist[currIndex]).removeClass('on');
			$(tablist[index]).addClass('on');
			
			$(panellist[currIndex]).css('display','none');
			$(panellist[index]).css('display','block');
			currIndex = index;
		});
		
		//滚动
		var currImage = 0,
			leftImage = 0,
			container = $('.ctrl .list ul');
		container.each(function(index,elem){
			var buff = [],n;
			for(n=0;n<imageList.length;++n){
				buff.push('<li><a '+(n==0?'class="on" ':'')+'href="#" onclick="return false;" index="'+n+'"><img alt="" src="'+imageList[n].smallURL+'" /></a></li>');
			}
			leftImage = 0;
			$(elem).css('width',140*imageList.length+'px')
			$(elem).html(buff.join(''));			
			$('.ctrl .prev_off').css('opacity','0.5');
		});
		$('.ctrl .next').click(function(event){
			var targetLeftIndex = (leftImage + 1),
				targetLeft;
				if(targetLeftIndex>=imageList.length-6){
					targetLeftIndex=imageList.length-6;
					$('.ctrl .next').css('opacity','0.5');
				}
				$('.ctrl .prev_off').css('opacity','');
				targetLeft = targetLeftIndex * 138;
				$('.ctrl .list').animate({scrollLeft:targetLeft},'slow',
					function(){
						$('.ctrl .list').scrollLeft(targetLeft);
					});
				leftImage = targetLeftIndex;
			return false;
		});
		$('.ctrl .prev_off').click(function(event){
			var targetLeftIndex = (leftImage - 1),
				targetLeft;
				if(targetLeftIndex<=0){
					targetLeftIndex=0;
					$('.ctrl .prev_off').css('opacity','0.5');
				}
				$('.ctrl .next').css('opacity','');
				targetLeft = targetLeftIndex * 138;
				$('.ctrl .list').animate({scrollLeft:targetLeft},'slow',
					function(){
						$('.ctrl .list').scrollLeft(targetLeft);
					});
				leftImage = targetLeftIndex;
			return false;
		});
		container.mouseover(function(event){
			var nodeName = event.target.nodeName;
			if(nodeName == 'IMG'||nodeName == 'img'){
				var index = $(event.target).parent().attr('index');
				container.find('.on').removeClass('on');
				$(event.target).parent().addClass('on');
				$('.still .view img').attr('src',imageList[index].bigURL);
			}
		});
		
	</script>
